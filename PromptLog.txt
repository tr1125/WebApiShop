public class UserRepositoryUnitTesting
{
[Fact]
public async Task GetUser_ValidCredentials_ReturnsUser()
{
// Arrange
var user = new User { FirstName= "Miriam", Email = "test@example.com", Password = "password" };

var mockContext = new Mock<MyShopContext>();
var users = new List<User>() {user };
mockContext. Setup(x => x.Users).ReturnsDbSet(users);

var userRepository = new UserRepository(mockContext.Object);
// Act
var result = await userRepository.getUser(user.Email, user.Password);

// Assert
Assert.Equal(user, result); })
